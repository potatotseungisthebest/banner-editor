            e.preventDefault();

            fileUploadArea.classList.remove('dragover');

            const files = e.dataTransfer.files;

            if (files.length > 0) {

                handleImageUpload(files[0]);

            }

        });


        imageUpload.addEventListener('change', (e) => {

            if (e.target.files.length > 0) {

                handleImageUpload(e.target.files[0]);

            }

        });


        // Handle image upload

        function handleImageUpload(file) {

            if (!file.type.startsWith('image/')) {

                alert('Please upload a valid image file.');

                return;

            }


            const reader = new FileReader();

            reader.onload = (e) => {

                bannerImage.src = e.target.result;

                

                // Show the image regardless of size selection

                bannerImage.style.display = 'block';

                

                // Enable download only if size is also selected

                if (bannerSize.value) {

                    downloadBtn.disabled = false;

                }

            };

            reader.readAsDataURL(file);

        }

            smallHeaderDisplay.textContent = e.target.value || 'Small Header Text';

        });

            bigHeaderDisplay.textContent = e.target.value || 'Main Headline';

        });


        description.addEventListener('input', (e) => {

            descriptionDisplay.textContent = e.target.value || 'Description';

        });


        cta1.addEventListener('input', (e) => {

            cta1Display.textContent = e.target.value || 'CTA Button Text';

        });


        // Image position handler

        imagePosition.addEventListener('change', (e) => {

            bannerImage.style.objectPosition = e.target.value;

        });


        // Banner size handler

        bannerSize.addEventListener('change', (e) => {

            if (e.target.value) {

                bannerPreview.className = 'banner-container ' + e.target.value + '-size';

                // Show banner content when size is selected

                const bannerContent = bannerPreview.querySelector('.banner-content');

                const gradientOverlay = bannerPreview.querySelector('.gradient-overlay');

                

                if (bannerContent) bannerContent.style.display = 'block';

                if (gradientOverlay) gradientOverlay.style.display = 'block';

                

                // Always show image if it exists

                if (bannerImage.src) {

                    bannerImage.style.display = 'block';

                }

                

                // Enable download if image is also uploaded

                if (bannerImage.src && !bannerImage.src.includes('data:image/svg+xml')) {

                    downloadBtn.disabled = false;

                }

            } else {

                bannerPreview.className = 'banner-container';

                // Hide banner content when no size selected, but keep image visible

                const bannerContent = bannerPreview.querySelector('.banner-content');

                const gradientOverlay = bannerPreview.querySelector('.gradient-overlay');

                

                if (bannerContent) bannerContent.style.display = 'none';

                if (gradientOverlay) gradientOverlay.style.display = 'none';

                

                // Keep image visible even without size selection

                if (bannerImage.src && !bannerImage.src.includes('data:image/svg+xml')) {

                    bannerImage.style.display = 'block';

                }

                

                downloadBtn.disabled = true;

            }

        });


        // Download functionality with exact preview matching

        downloadBtn.addEventListener('click', async () => {

            downloadBtn.textContent = '⏳ Generating...';

            downloadBtn.disabled = true;


            try {

                const selectedSize = bannerSize.value;

                const dimensions = selectedSize === 'desktop' ? 

                    { width: 2700, height: 900 } : 

                    { width: 768, height: 600 };


                // Calculate scale factor from preview to final size

                const previewDimensions = selectedSize === 'desktop' ? 

                    { width: 900, height: 300 } : 

                    { width: 384, height: 300 };

                

                const scaleX = dimensions.width / previewDimensions.width;

                const scaleY = dimensions.height / previewDimensions.height;


                // Create a high-resolution canvas

                const canvas = document.createElement('canvas');

                canvas.width = dimensions.width;

                canvas.height = dimensions.height;

                const ctx = canvas.getContext('2d');


                // Draw the background image

                const img = new Image();

                img.crossOrigin = 'anonymous';

                

                await new Promise((resolve, reject) => {

                    img.onload = resolve;

                    img.onerror = reject;

                    img.src = bannerImage.src;

                });


                // Draw image with object-fit: cover behavior

                const drawImageCover = (ctx, img, x, y, w, h) => {

                    const imgRatio = img.width / img.height;

                    const canvasRatio = w / h;

                    let drawWidth, drawHeight, offsetX = 0, offsetY = 0;


                    if (imgRatio > canvasRatio) {

                        drawHeight = h;

                        drawWidth = h * imgRatio;

                    } else {

                        drawWidth = w;

                        drawHeight = w / imgRatio;

                    }


                    // Apply object-position

                    const position = imagePosition.value || 'center';

                    const positions = position.split(' ');

                    

                    // Horizontal positioning

                    if (positions.includes('left')) {

                        offsetX = 0;

                    } else if (positions.includes('right')) {

                        offsetX = w - drawWidth;

                    } else {

                        offsetX = (w - drawWidth) / 2;

                    }


                    // Vertical positioning

                    if (positions.includes('top')) {

                        offsetY = 0;

                    } else if (positions.includes('bottom')) {

                        offsetY = h - drawHeight;

                    } else {

                        offsetY = (h - drawHeight) / 2;

                    }


                    ctx.drawImage(img, offsetX, offsetY, drawWidth, drawHeight);

                };


                drawImageCover(ctx, img, 0, 0, dimensions.width, dimensions.height);


                // Draw gradient overlay

                if (selectedSize === 'desktop') {

                    const gradient = ctx.createLinearGradient(0, 0, dimensions.width * 0.6, 0);

                    gradient.addColorStop(0, 'rgba(0,0,0,0.8)');

                    gradient.addColorStop(0.5, 'rgba(0,0,0,0.5)');

                    gradient.addColorStop(1, 'rgba(0,0,0,0)');

                    ctx.fillStyle = gradient;

                    ctx.fillRect(0, 0, dimensions.width * 0.6, dimensions.height);

                } else {

                    // Mobile: bottom gradient

                    const gradient = ctx.createLinearGradient(0, dimensions.height, 0, dimensions.height * 0.3);

                    gradient.addColorStop(0, 'rgba(0,0,0,0.9)');

                    gradient.addColorStop(0.6, 'rgba(0,0,0,0.6)');

                    gradient.addColorStop(1, 'rgba(0,0,0,0)');

                    ctx.fillStyle = gradient;

                    ctx.fillRect(0, dimensions.height * 0.3, dimensions.width, dimensions.height * 0.7);

                }


                // Set up text styles

                ctx.fillStyle = 'white';

                ctx.textBaseline = 'top';

                ctx.shadowColor = 'rgba(0, 0, 0, 0.5)';

                ctx.shadowBlur = 3;

                ctx.shadowOffsetX = 1;

                ctx.shadowOffsetY = 1;


                if (selectedSize === 'desktop') {

                    // Desktop layout - left aligned

                    const leftMargin = dimensions.width * 0.05;

                    

                    // Calculate total content height to center it properly

                    const smallHeaderHeight = 60;

                    const bigHeaderHeight = 110;

                    const descriptionHeight = 50;

                    const buttonHeight = 70;

                    const spacing = 40;

                    const totalContentHeight = smallHeaderHeight + bigHeaderHeight + descriptionHeight + buttonHeight + (spacing * 3);

                    

                    let currentY = (dimensions.height - totalContentHeight) / 2;


                    // Small header

                    ctx.font = `600 ${60}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    ctx.fillText(smallHeaderDisplay.textContent, leftMargin, currentY);

                    currentY += smallHeaderHeight + 20;


                    // Big header

                    ctx.font = `700 ${110}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    const bigHeaderLines = wrapText(ctx, bigHeaderDisplay.textContent, dimensions.width * 0.5);

                    bigHeaderLines.forEach(line => {

                        ctx.fillText(line, leftMargin, currentY);

                        currentY += 120;

                    });

                    currentY += spacing;


                    // Description

                    ctx.font = `400 ${50}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    const descLines = wrapText(ctx, descriptionDisplay.textContent, dimensions.width * 0.4);

                    descLines.forEach(line => {

                        ctx.fillText(line, leftMargin, currentY);

                        currentY += 65;

                    });

                    currentY += spacing + 20;


                    // CTA Button

                    ctx.shadowBlur = 0;

                    const buttonPadding = { x: 90, y: 35 };

                    ctx.font = `600 ${45}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    const buttonText = cta1Display.textContent;

                    const buttonWidth = ctx.measureText(buttonText).width + (buttonPadding.x * 2);

                    const buttonHeightFinal = 55 + (buttonPadding.y * 2);

                    

                    ctx.fillStyle = '#1a73e8';

                    roundRect(ctx, leftMargin, currentY, buttonWidth, buttonHeightFinal, 15);

                    ctx.fill();

                    

                    ctx.fillStyle = 'white';

                    ctx.fillText(buttonText, leftMargin + buttonPadding.x, currentY + buttonPadding.y);

                } else {

                    // Mobile layout - center bottom aligned

                    ctx.textAlign = 'center';

                    const centerX = dimensions.width / 2;

                    const bottomMargin = dimensions.height * 0.05;

                    let currentY = dimensions.height - bottomMargin;


                    // Work backwards from bottom

                    // CTA Button

                    ctx.shadowBlur = 0;

                    const buttonPadding = { x: 40, y: 16 };

                    ctx.font = `600 ${18}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    const buttonText = cta1Display.textContent;

                    const buttonWidth = ctx.measureText(buttonText).width + (buttonPadding.x * 2);

                    const buttonHeight = 24 + (buttonPadding.y * 2);

                    

                    currentY -= buttonHeight;

                    ctx.fillStyle = '#1a73e8';

                    roundRect(ctx, centerX - buttonWidth/2, currentY, buttonWidth, buttonHeight, 8);

                    ctx.fill();

                    

                    ctx.fillStyle = 'white';

                    ctx.fillText(buttonText, centerX, currentY + buttonPadding.y);

                    

                    currentY -= 30;

                    ctx.shadowBlur = 3;


                    // Description

                    ctx.font = `400 ${20}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    const descLines = wrapText(ctx, descriptionDisplay.textContent, dimensions.width * 0.85);

                    currentY -= descLines.length * 28;

                    descLines.forEach(line => {

                        ctx.fillText(line, centerX, currentY);

                        currentY += 28;

                    });

                    currentY -= descLines.length * 28 + 20;


                    // Big header

                    ctx.font = `700 ${36}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    const bigHeaderLines = wrapText(ctx, bigHeaderDisplay.textContent, dimensions.width * 0.85);

                    currentY -= bigHeaderLines.length * 44;

                    bigHeaderLines.forEach(line => {

                        ctx.fillText(line, centerX, currentY);

                        currentY += 44;

                    });

                    currentY -= bigHeaderLines.length * 44 + 15;


                    // Small header

                    ctx.font = `600 ${18}px -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif`;

                    currentY -= 24;

                    ctx.fillText(smallHeaderDisplay.textContent, centerX, currentY);

                }


                // Convert to blob and download

                canvas.toBlob((blob) => {

                    const url = URL.createObjectURL(blob);

                    const link = document.createElement('a');

                    link.download = `potato-banner-${selectedSize}-${dimensions.width}x${dimensions.height}.webp`;

                    link.href = url;

                    document.body.appendChild(link);

                    link.click();

                    document.body.removeChild(link);

                    URL.revokeObjectURL(url);

                }, 'image/webp', 0.95);


            } catch (error) {

                console.error('Error generating image:', error);

                alert('Error generating image. Please try again.');

            } finally {

                downloadBtn.textContent = 'Download Banner';

                downloadBtn.disabled = false;

            }

        });


        // Helper function to wrap text

        function wrapText(ctx, text, maxWidth) {

            const words = text.split(' ');

            const lines = [];

            let currentLine = '';


            for (let word of words) {

                const testLine = currentLine ? currentLine + ' ' + word : word;

                const metrics = ctx.measureText(testLine);

                

                if (metrics.width > maxWidth && currentLine) {

                    lines.push(currentLine);

                    currentLine = word;

                } else {

                    currentLine = testLine;

                }

            }

            

            if (currentLine) {

                lines.push(currentLine);

            }

            

            return lines;

        }


        // Helper function to draw rounded rectangle

        function roundRect(ctx, x, y, width, height, radius) {

            ctx.beginPath();

            ctx.moveTo(x + radius, y);

            ctx.lineTo(x + width - radius, y);

            ctx.quadraticCurveTo(x + width, y, x + width, y + radius);

            ctx.lineTo(x + width, y + height - radius);

            ctx.quadraticCurveTo(x + width, y + height, x + width - radius, y + height);

            ctx.lineTo(x + radius, y + height);

            ctx.quadraticCurveTo(x, y + height, x, y + height - radius);

            ctx.lineTo(x, y + radius);

            ctx.quadraticCurveTo(x, y, x + radius, y);

            ctx.closePath();

        }
